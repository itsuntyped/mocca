# Mocca - runtime dependencies.
# Pinned with lower bounds. Tested on Python 3.11–3.12 (see note below).

fastapi>=0.110          # Web framework / API layer
uvicorn[standard]>=0.27 # ASGI server (the [standard] extra adds websockets, http-tools, etc.)
httpx>=0.27             # Async HTTP client used to download models from Hugging Face
jinja2>=3.1             # HTML templating for the single served page
python-multipart>=0.0.9 # Form parsing
pydantic>=2.6           # Request/response models

# Reads YouTube video transcripts for the youtube_transcript tool (a network
# tool, gated behind the web-search toggle). Pure Python, no API key. Imported
# lazily so a missing install never breaks startup.
youtube-transcript-api>=1.0

# Identifies the carrier for a tracking number and builds the carrier's official
# tracking URL for the track_shipment tool. Pure Python, no API key, no network
# (it parses the number against published carrier formats). Imported lazily so a
# missing install never breaks startup.
tracking-numbers>=0.1.8

# Local inference engine - runs GGUF models in-process (no external server).
# Mocca imports this lazily, so the UI and model downloads still work even if it
# fails to install; but you need it to actually chat.
#
# NOTE: prebuilt wheels exist for common platforms on Python 3.11–3.12. On very
# new Python versions (e.g. 3.14) a wheel may not be published yet, forcing a
# source build (which needs CMake + a C compiler). If install fails, either use
# Python 3.11/3.12 or install a prebuilt wheel - see the project readme.
#
# This is the CPU build. For NVIDIA/CUDA GPU acceleration, run
# 'python scripts/setup.py' instead - it detects the GPU and swaps in the
# prebuilt CUDA wheel plus the nvidia-*-cu12 runtime. See the readme's
# "GPU acceleration" section.
llama-cpp-python>=0.2.79

# Hugging Face API with httpx (above); no client library or API key required.
